// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(walstrobremraap_voortplanting_bwk, bwk(2130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(walstrobremraap_voortplanting_cluster, fuzzyClusterIdU(walstrobremraap_voortplanting_bwk, 20, meter))

deleteVariable(walstrobremraap_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(walstrobremraap_leefgebied, cFilterGeArea(walstrobremraap_voortplanting_cluster, 0.16, hectare))
deleteVariable(walstrobremraap_voortplanting_cluster)

write('ANB2019_Walstrobremraap2018.asc', IsNonZero(walstrobremraap_leefgebied))
deleteVariable(walstrobremraap_leefgebied)
