// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(ongelijkbladigfonteinkruid_voortplanting_bwk, bwk(3130%, 3260%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(ongelijkbladigfonteinkruid_voortplanting_cluster, fuzzyClusterIdU(ongelijkbladigfonteinkruid_voortplanting_bwk, 20, meter))

deleteVariable(ongelijkbladigfonteinkruid_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(ongelijkbladigfonteinkruid_leefgebied, cFilterGeArea(ongelijkbladigfonteinkruid_voortplanting_cluster, 0.16, hectare))
deleteVariable(ongelijkbladigfonteinkruid_voortplanting_cluster)

write('ANB2019_OngelijkbladigFonteinkruid2018.asc', IsNonZero(ongelijkbladigfonteinkruid_leefgebied))
deleteVariable(ongelijkbladigfonteinkruid_leefgebied)
