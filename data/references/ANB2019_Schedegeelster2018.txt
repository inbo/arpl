// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(schedegeelster_voortplanting_bwk, bwk(9130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(schedegeelster_voortplanting_cluster, fuzzyClusterIdU(schedegeelster_voortplanting_bwk, 20, meter))

deleteVariable(schedegeelster_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(schedegeelster_leefgebied, cFilterGeArea(schedegeelster_voortplanting_cluster, 0.16, hectare))
deleteVariable(schedegeelster_voortplanting_cluster)

write('ANB2019_Schedegeelster2018.asc', IsNonZero(schedegeelster_leefgebied))
deleteVariable(schedegeelster_leefgebied)
