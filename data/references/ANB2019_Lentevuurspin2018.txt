// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(lentevuurspin_voortplanting_bwk, bwk(2310%, 2330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(lentevuurspin_voortplanting_cluster, fuzzyClusterIdU(lentevuurspin_voortplanting_bwk, 20, meter))

deleteVariable(lentevuurspin_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 1ha groot zijn

assign(lentevuurspin_leefgebied, cFilterGeArea(lentevuurspin_voortplanting_cluster, 2, hectare))
deleteVariable(lentevuurspin_voortplanting_cluster)

write('ANB2019_Lentevuurspin2018.asc', IsNonZero(lentevuurspin_leefgebied))
