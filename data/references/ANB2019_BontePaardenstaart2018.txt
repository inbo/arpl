// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(bontepaardenstaart_voortplanting_bwk, bwk(2190%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(bontepaardenstaart_voortplanting_cluster, fuzzyClusterIdU(bontepaardenstaart_voortplanting_bwk, 20, meter))

deleteVariable(bontepaardenstaart_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(bontepaardenstaart_leefgebied, cFilterGeArea(bontepaardenstaart_voortplanting_cluster, 0.16, hectare))
deleteVariable(bontepaardenstaart_voortplanting_cluster)

write('ANB2019_BontePaardenstaart2018.asc', IsNonZero(bontepaardenstaart_leefgebied))
deleteVariable(bontepaardenstaart_leefgebied)
