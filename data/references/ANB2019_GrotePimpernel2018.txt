// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(grotepimpernel_voortplanting_bwk, bwk(6510%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(grotepimpernel_voortplanting_cluster, fuzzyClusterIdU(grotepimpernel_voortplanting_bwk, 20, meter))

deleteVariable(grotepimpernel_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(grotepimpernel_leefgebied, cFilterGeArea(grotepimpernel_voortplanting_cluster, 0.16, hectare))
deleteVariable(grotepimpernel_voortplanting_cluster)

write('ANB2019_GrotePimpernel2018.asc', IsNonZero(grotepimpernel_leefgebied))
deleteVariable(grotepimpernel_leefgebied)
