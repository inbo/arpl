// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kikkerbeet_voortplanting_bwk, bwk(3150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kikkerbeet_voortplanting_cluster, fuzzyClusterIdU(kikkerbeet_voortplanting_bwk, 20, meter))

deleteVariable(kikkerbeet_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(kikkerbeet_leefgebied, cFilterGeArea(kikkerbeet_voortplanting_cluster, 0.16, hectare))
deleteVariable(kikkerbeet_voortplanting_cluster)

write('ANB2019_Kikkerbeet2018.asc', IsNonZero(kikkerbeet_leefgebied))
deleteVariable(kikkerbeet_leefgebied)
