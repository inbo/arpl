// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(veenmosorchis_voortplanting_bwk, bwk(7140%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(veenmosorchis_voortplanting_cluster, fuzzyClusterIdU(veenmosorchis_voortplanting_bwk, 20, meter))

deleteVariable(veenmosorchis_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(veenmosorchis_leefgebied, cFilterGeArea(veenmosorchis_voortplanting_cluster, 0.16, hectare))
deleteVariable(veenmosorchis_voortplanting_cluster)

write('ANB2019_Veenmosorchis2018.asc', IsNonZero(veenmosorchis_leefgebied))
deleteVariable(veenmosorchis_leefgebied)
