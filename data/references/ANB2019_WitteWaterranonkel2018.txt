// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(wittewaterranonkel_voortplanting_bwk, bwk(3130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(wittewaterranonkel_voortplanting_cluster, fuzzyClusterIdU(wittewaterranonkel_voortplanting_bwk, 20, meter))

deleteVariable(wittewaterranonkel_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(wittewaterranonkel_leefgebied, cFilterGeArea(wittewaterranonkel_voortplanting_cluster, 0.16, hectare))
deleteVariable(wittewaterranonkel_voortplanting_cluster)

write('ANB2019_WitteWaterranonkel2018.asc', IsNonZero(wittewaterranonkel_leefgebied))
deleteVariable(wittewaterranonkel_leefgebied)
