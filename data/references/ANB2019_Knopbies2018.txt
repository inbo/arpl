// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(knopbies_voortplanting_bwk, bwk(2190%, 7230%, mk%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(knopbies_voortplanting_cluster, fuzzyClusterIdU(knopbies_voortplanting_bwk, 20, meter))

deleteVariable(knopbies_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(knopbies_leefgebied, cFilterGeArea(knopbies_voortplanting_cluster, 0.16, hectare))
deleteVariable(knopbies_voortplanting_cluster)

write('ANB2019_Knopbies2018.asc', IsNonZero(knopbies_leefgebied))
deleteVariable(knopbies_leefgebied)
