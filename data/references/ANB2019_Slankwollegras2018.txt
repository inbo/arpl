// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(slankwollegras_voortplanting_bwk, bwk(7140%, 7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(slankwollegras_voortplanting_cluster, fuzzyClusterIdU(slankwollegras_voortplanting_bwk, 20, meter))

deleteVariable(slankwollegras_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(slankwollegras_leefgebied, cFilterGeArea(slankwollegras_voortplanting_cluster, 0.16, hectare))
deleteVariable(slankwollegras_voortplanting_cluster)

write('ANB2019_SlankWollegras2018.asc', IsNonZero(slankwollegras_leefgebied))
deleteVariable(slankwollegras_leefgebied)
