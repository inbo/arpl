// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(veenpluis_voortplanting_bwk, bwk(3160%, 4010%, 7110%, 7140%, 7150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(veenpluis_voortplanting_cluster, fuzzyClusterIdU(veenpluis_voortplanting_bwk, 20, meter))

deleteVariable(veenpluis_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(veenpluis_leefgebied, cFilterGeArea(veenpluis_voortplanting_cluster, 0.16, hectare))
deleteVariable(veenpluis_voortplanting_cluster)

write('ANB2019_Veenpluis2018.asc', IsNonZero(veenpluis_leefgebied))
deleteVariable(veenpluis_leefgebied)
