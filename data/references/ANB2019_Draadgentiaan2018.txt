// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(draadgentiaan_voortplanting_bwk, bwk(3130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(draadgentiaan_voortplanting_cluster, fuzzyClusterIdU(draadgentiaan_voortplanting_bwk, 20, meter))

deleteVariable(draadgentiaan_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(draadgentiaan_leefgebied, cFilterGeArea(draadgentiaan_voortplanting_cluster, 0.16, hectare))
deleteVariable(draadgentiaan_voortplanting_cluster)

write('ANB2019_Draadgentiaan2018.asc', IsNonZero(draadgentiaan_leefgebied))
deleteVariable(draadgentiaan_leefgebied)
