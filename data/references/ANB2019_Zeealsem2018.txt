// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(zeealsem_voortplanting_bwk, bwk(1330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(zeealsem_voortplanting_cluster, fuzzyClusterIdU(zeealsem_voortplanting_bwk, 20, meter))

deleteVariable(zeealsem_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(zeealsem_leefgebied, cFilterGeArea(zeealsem_voortplanting_cluster, 0.16, hectare))
deleteVariable(zeealsem_voortplanting_cluster)

write('ANB2019_Zeealsem2018.asc', IsNonZero(zeealsem_leefgebied))
deleteVariable(zeealsem_leefgebied)
