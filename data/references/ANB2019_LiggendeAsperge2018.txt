// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(liggendeasperge_voortplanting_bwk, bwk(2130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(liggendeasperge_voortplanting_cluster, fuzzyClusterIdU(liggendeasperge_voortplanting_bwk, 20, meter))

deleteVariable(liggendeasperge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(liggendeasperge_leefgebied, cFilterGeArea(liggendeasperge_voortplanting_cluster, 0.16, hectare))
deleteVariable(liggendeasperge_voortplanting_cluster)

write('ANB2019_LiggendeAsperge2018.asc', IsNonZero(liggendeasperge_leefgebied))
deleteVariable(liggendeasperge_leefgebied)
