// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(heidekartelblad_voortplanting_bwk, bwk(4010%, 6230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(heidekartelblad_voortplanting_cluster, fuzzyClusterIdU(heidekartelblad_voortplanting_bwk, 20, meter))

deleteVariable(heidekartelblad_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(heidekartelblad_leefgebied, cFilterGeArea(heidekartelblad_voortplanting_cluster, 0.16, hectare))
deleteVariable(heidekartelblad_voortplanting_cluster)

write('ANB2019_Heidekartelblad2018.asc', IsNonZero(heidekartelblad_leefgebied))
deleteVariable(heidekartelblad_leefgebied)
