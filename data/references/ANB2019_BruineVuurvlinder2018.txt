// baken geschikte biotopen af
assign(bruinevuurvlinder_bwk, bwk(hu%, ha%, hn%, hp+, hr+, 6510%, rbbkam%, rbbvos%))

// geschikte biotopen mogen maximaal op een onderlinge afstand van 50 meter liggen
assign(bruinevuurvlinder_bwk_cluster, fuzzyClusterIdU(bruinevuurvlinder_bwk, 50, meter))
deleteVariable(bruinevuurvlinder_bwk)

// leefgebied moet minstens 1 hectare groot zijn
assign(bruinevuurvlinder_leefgebied, cFilterGeArea(bruinevuurvlinder_bwk_cluster, 5, hectare))
deleteVariable(bruinevuurvlinder_bwk_cluster)

// 0/1 kaarten maken
write('ANB2019_bruinevuurvlinder2018.asc', IsNonZero(bruinevuurvlinder_leefgebied))
deleteVariable(bruinevuurvlinder_leefgebied)
