// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(lavendelhei_voortplanting_bwk, bwk(7110%, 7120%, 7140%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(lavendelhei_voortplanting_cluster, fuzzyClusterIdU(lavendelhei_voortplanting_bwk, 20, meter))

deleteVariable(lavendelhei_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(lavendelhei_leefgebied, cFilterGeArea(lavendelhei_voortplanting_cluster, 0.16, hectare))
deleteVariable(lavendelhei_voortplanting_cluster)

write('ANB2019_Lavendelhei2018.asc', IsNonZero(lavendelhei_leefgebied))
deleteVariable(lavendelhei_leefgebied)
