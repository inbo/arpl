// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(koprus_voortplanting_bwk, bwk(3130_na%, b+))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(koprus_voortplanting_cluster, fuzzyClusterIdU(koprus_voortplanting_bwk, 20, meter))

deleteVariable(koprus_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(koprus_leefgebied, cFilterGeArea(koprus_voortplanting_cluster, 0.16, hectare))
deleteVariable(koprus_voortplanting_cluster)

write('ANB2019_Koprus2018.asc', IsNonZero(koprus_leefgebied))
deleteVariable(koprus_leefgebied)
