// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kleinblaasjeskruid_voortplanting_bwk, bwk(3110%, 3160%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kleinblaasjeskruid_voortplanting_cluster, fuzzyClusterIdU(kleinblaasjeskruid_voortplanting_bwk, 20, meter))

deleteVariable(kleinblaasjeskruid_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(kleinblaasjeskruid_leefgebied, cFilterGeArea(kleinblaasjeskruid_voortplanting_cluster, 0.16, hectare))
deleteVariable(kleinblaasjeskruid_voortplanting_cluster)

write('ANB2019_KleinBlaasjeskruid2018.asc', IsNonZero(kleinblaasjeskruid_leefgebied))
deleteVariable(kleinblaasjeskruid_leefgebied)
