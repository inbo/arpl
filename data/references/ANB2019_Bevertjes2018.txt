// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(bevertjes_voortplanting_bwk, bwk(2130%, 2190%, 6230%, 6510%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(bevertjes_voortplanting_cluster, fuzzyClusterIdU(bevertjes_voortplanting_bwk, 20, meter))

deleteVariable(bevertjes_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 1ha groot zijn

assign(bevertjes_leefgebied, cFilterGeArea(bevertjes_voortplanting_cluster, 1, hectare))
deleteVariable(bevertjes_voortplanting_cluster)

write('ANB2019_Bevertjes2018.asc', IsNonZero(bevertjes_leefgebied))
deleteVariable(bevertjes_leefgebied)
