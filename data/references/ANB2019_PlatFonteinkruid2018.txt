// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(platfonteinkruid_voortplanting_bwk, bwk(3150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(platfonteinkruid_voortplanting_cluster, fuzzyClusterIdU(platfonteinkruid_voortplanting_bwk, 20, meter))

deleteVariable(platfonteinkruid_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(platfonteinkruid_leefgebied, cFilterGeArea(platfonteinkruid_voortplanting_cluster, 0.16, hectare))
deleteVariable(platfonteinkruid_voortplanting_cluster)

write('ANB2019_PlatFonteinkruid2018.asc', IsNonZero(platfonteinkruid_leefgebied))
deleteVariable(platfonteinkruid_leefgebied)
