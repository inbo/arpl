// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(hondsviooltje_voortplanting_bwk, bwk(2130%, 6230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(hondsviooltje_voortplanting_cluster, fuzzyClusterIdU(hondsviooltje_voortplanting_bwk, 20, meter))

deleteVariable(hondsviooltje_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(hondsviooltje_leefgebied, cFilterGeArea(hondsviooltje_voortplanting_cluster, 0.16, hectare))
deleteVariable(hondsviooltje_voortplanting_cluster)

write('ANB2019_Hondsviooltje2018.asc', IsNonZero(hondsviooltje_leefgebied))
deleteVariable(hondsviooltje_leefgebied)
