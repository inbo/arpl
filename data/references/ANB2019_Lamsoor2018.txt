// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(lamsoor_voortplanting_bwk, bwk(1130%, 1310%, 1330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(lamsoor_voortplanting_cluster, fuzzyClusterIdU(lamsoor_voortplanting_bwk, 20, meter))

deleteVariable(lamsoor_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(lamsoor_leefgebied, cFilterGeArea(lamsoor_voortplanting_cluster, 0.16, hectare))
deleteVariable(lamsoor_voortplanting_cluster)

write('ANB2019_Lamsoor2018.asc', IsNonZero(lamsoor_leefgebied))
deleteVariable(lamsoor_leefgebied)
