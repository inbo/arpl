// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(schubzegge_voortplanting_bwk, bwk(7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(schubzegge_voortplanting_cluster, fuzzyClusterIdU(schubzegge_voortplanting_bwk, 20, meter))

deleteVariable(schubzegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(schubzegge_leefgebied, cFilterGeArea(schubzegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(schubzegge_voortplanting_cluster)

write('ANB2019_Schubzegge2018.asc', IsNonZero(schubzegge_leefgebied))
deleteVariable(schubzegge_leefgebied)
