// leefgebied moet minstens 5 hectare groot zijn
assign(hazelworm_leefgebied, cFilterGeArea(hazelworm_bwk_cluster, 5, hectare))

assign(hazelworm_leefgebied_ecodistrict, and(hazelworm_leefgebied, hazelworm_ecodistrict))
deleteVariable(hazelworm_ecodistrict)
deleteVariable(hazelworm_leefgebied)
