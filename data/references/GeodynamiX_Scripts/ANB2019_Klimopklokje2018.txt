// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(klimopklokje_voortplanting_bwk, bwk(6410%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(klimopklokje_voortplanting_cluster, fuzzyClusterIdU(klimopklokje_voortplanting_bwk, 20, meter))

deleteVariable(klimopklokje_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(klimopklokje_leefgebied, cFilterGeArea(klimopklokje_voortplanting_cluster, 0.16, hectare))
deleteVariable(klimopklokje_voortplanting_cluster)

write('ANB2019_Klimopklokje2018.asc', IsNonZero(klimopklokje_leefgebied))
deleteVariable(klimopklokje_leefgebied)
