// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(dwergvlas_voortplanting_bwk, bwk(3130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(dwergvlas_voortplanting_cluster, fuzzyClusterIdU(dwergvlas_voortplanting_bwk, 20, meter))

deleteVariable(dwergvlas_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(dwergvlas_leefgebied, cFilterGeArea(dwergvlas_voortplanting_cluster, 0.16, hectare))
deleteVariable(dwergvlas_voortplanting_cluster)

write('ANB2019_Dwergvlas2018.asc', IsNonZero(dwergvlas_leefgebied))
deleteVariable(dwergvlas_leefgebied)
