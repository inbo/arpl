// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(waterlobelia_voortplanting_bwk, bwk(3110%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(waterlobelia_voortplanting_cluster, fuzzyClusterIdU(waterlobelia_voortplanting_bwk, 20, meter))

deleteVariable(waterlobelia_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(waterlobelia_leefgebied, cFilterGeArea(waterlobelia_voortplanting_cluster, 0.16, hectare))
deleteVariable(waterlobelia_voortplanting_cluster)

write('ANB2019_Waterlobelia2018.asc', IsNonZero(waterlobelia_leefgebied))
deleteVariable(waterlobelia_leefgebied)
