// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kleinewolfsklauw_voortplanting_bwk, bwk(4030%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kleinewolfsklauw_voortplanting_cluster, fuzzyClusterIdU(kleinewolfsklauw_voortplanting_bwk, 20, meter))

deleteVariable(kleinewolfsklauw_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(kleinewolfsklauw_leefgebied, cFilterGeArea(kleinewolfsklauw_voortplanting_cluster, 0.16, hectare))
deleteVariable(kleinewolfsklauw_voortplanting_cluster)

write('ANB2019_KleineWolfsklauw2018.asc', IsNonZero(kleinewolfsklauw_leefgebied))
deleteVariable(kleinewolfsklauw_leefgebied)
