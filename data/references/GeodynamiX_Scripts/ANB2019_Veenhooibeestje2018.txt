// baken geschikte biotopen af
assign(veenhooibeestje_bwk, bwk(7110%, ces+, ces, ces-, t, t-, t+))

// geschikte biotopen mogen maximaal op een onderlinge afstand van 50 meter liggen
assign(veenhooibeestje_bwk_cluster, fuzzyClusterIdU(veenhooibeestje_bwk, 50, meter))
deleteVariable(veenhooibeestje_bwk)

// leefgebied moet minstens 5 hectare groot zijn
assign(veenhooibeestje_leefgebied, cFilterGeArea(veenhooibeestje_bwk_cluster, 5, hectare))

// beperk de leefgebiedenkaart tot bepaalde ecoregio's
assign(veenhooibeestje_ecodistrict, Ge('ecoregio.asc', 1))
assign(veenhooibeestje_leefgebied_ecodistrict, and(veenhooibeestje_leefgebied, veenhooibeestje_ecodistrict))
deleteVariable(veenhooibeestje_leefgebied)
deleteVariable(veenhooibeestje_ecodistrict)

// individuele nummers geven aan clusters
//write('veenhooibeestje_leefgebied_ecodistrict_id.asc', if(veenhooibeestje_leefgebied_ecodistrict, veenhooibeestje_bwk_cluster, 0))
deleteVariable(veenhooibeestje_bwk_cluster)

// 0/1 kaarten maken
write('ANB2019_Veenhooibeestje2018.asc', IsNonZero(veenhooibeestje_leefgebied_ecodistrict))
deleteVariable(veenhooibeestje_leefgebied_ecodistrict)
