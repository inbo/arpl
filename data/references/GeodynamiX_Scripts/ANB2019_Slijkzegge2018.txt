// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(slijkzegge_voortplanting_bwk, bwk(3160%, 7110%, 7140%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(slijkzegge_voortplanting_cluster, fuzzyClusterIdU(slijkzegge_voortplanting_bwk, 20, meter))

deleteVariable(slijkzegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(slijkzegge_leefgebied, cFilterGeArea(slijkzegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(slijkzegge_voortplanting_cluster)

write('ANB2019_Slijkzegge2018.asc', IsNonZero(slijkzegge_leefgebied))
deleteVariable(slijkzegge_leefgebied)
