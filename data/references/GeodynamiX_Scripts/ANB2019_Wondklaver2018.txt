// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(wondklaver_voortplanting_bwk, bwk(2130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(wondklaver_voortplanting_cluster, fuzzyClusterIdU(wondklaver_voortplanting_bwk, 20, meter))

deleteVariable(wondklaver_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(wondklaver_leefgebied, cFilterGeArea(wondklaver_voortplanting_cluster, 0.16, hectare))
deleteVariable(wondklaver_voortplanting_cluster)

write('ANB2019_Wondklaver2018.asc', IsNonZero(wondklaver_leefgebied))
deleteVariable(wondklaver_leefgebied)
