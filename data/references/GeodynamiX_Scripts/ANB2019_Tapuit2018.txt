// clustering: indien geschikte cel binnen 50 meter, dan hoort die cel bij het voortplantingscluster

// optimale voortplantingsclusters moeten minimaal 50 ha groot zijn
assign(tapuit_leefgebied, cFilterGeArea(tapuit_voortplanting_cluster, 50, hectare))

assign(tapuit_ecodistrict, Ge('ecoregio.asc', 1))
assign(tapuit_leefgebied_ecodistrict, and(tapuit_leefgebied, tapuit_ecodistrict))
deleteVariable(tapuit_leefgebied)

deleteVariable(tapuit_ecodistrict)
