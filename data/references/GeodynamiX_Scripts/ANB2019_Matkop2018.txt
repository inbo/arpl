// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(matkop_voortplanting_bwk, bwk(2180%, 91E0%, 91F0%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(matkop_voortplanting_cluster, fuzzyClusterIdU(matkop_voortplanting_bwk, 20, meter))

deleteVariable(matkop_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 5ha groot zijn

assign(matkop_leefgebied, cFilterGeArea(matkop_voortplanting_cluster, 5, hectare))
deleteVariable(matkop_voortplanting_cluster)

write('ANB2019_Matkop2018.asc', IsNonZero(matkop_leefgebied))
deleteVariable(matkop_leefgebied)
