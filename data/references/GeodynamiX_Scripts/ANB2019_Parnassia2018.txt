// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(parnassia_voortplanting_bwk, bwk(2190%, 6410%, 7140%, 7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(parnassia_voortplanting_cluster, fuzzyClusterIdU(parnassia_voortplanting_bwk, 20, meter))

deleteVariable(parnassia_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(parnassia_leefgebied, cFilterGeArea(parnassia_voortplanting_cluster, 0.16, hectare))
deleteVariable(parnassia_voortplanting_cluster)

write('ANB2019_Parnassia2018.asc', IsNonZero(parnassia_leefgebied))
deleteVariable(parnassia_leefgebied)
