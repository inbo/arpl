// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(stekendebies_voortplanting_bwk, bwk(1130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(stekendebies_voortplanting_cluster, fuzzyClusterIdU(stekendebies_voortplanting_bwk, 20, meter))

deleteVariable(stekendebies_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(stekendebies_leefgebied, cFilterGeArea(stekendebies_voortplanting_cluster, 0.16, hectare))
deleteVariable(stekendebies_voortplanting_cluster)

write('ANB2019_StekendeBies2018.asc', IsNonZero(stekendebies_leefgebied))
deleteVariable(stekendebies_leefgebied)
