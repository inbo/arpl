// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(wielewaal_voortplanting_bwk, bwk(2180%, 91E0%, 91F0%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(wielewaal_voortplanting_cluster, fuzzyClusterIdU(wielewaal_voortplanting_bwk, 20, meter))

deleteVariable(wielewaal_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 10ha groot zijn

assign(wielewaal_leefgebied, cFilterGeArea(wielewaal_voortplanting_cluster, 10, hectare))
deleteVariable(wielewaal_voortplanting_cluster)

write('ANB2019_Wielewaal2018.asc', IsNonZero(wielewaal_leefgebied))
deleteVariable(wielewaal_leefgebied)
