// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(bruinesnavelbies_voortplanting_bwk, bwk(3160%, 4010%, 7140%, 7150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(bruinesnavelbies_voortplanting_cluster, fuzzyClusterIdU(bruinesnavelbies_voortplanting_bwk, 20, meter))

deleteVariable(bruinesnavelbies_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(bruinesnavelbies_leefgebied, cFilterGeArea(bruinesnavelbies_voortplanting_cluster, 0.16, hectare))
deleteVariable(bruinesnavelbies_voortplanting_cluster)

write('ANB2019_BruineSnavelbies2018.asc', IsNonZero(bruinesnavelbies_leefgebied))
deleteVariable(bruinesnavelbies_leefgebied)
