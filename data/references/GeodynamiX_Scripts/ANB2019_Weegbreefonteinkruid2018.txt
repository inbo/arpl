// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(weegbreefonteinkruid_voortplanting_bwk, bwk(3140%, 3260%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(weegbreefonteinkruid_voortplanting_cluster, fuzzyClusterIdU(weegbreefonteinkruid_voortplanting_bwk, 20, meter))

deleteVariable(weegbreefonteinkruid_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(weegbreefonteinkruid_leefgebied, cFilterGeArea(weegbreefonteinkruid_voortplanting_cluster, 0.16, hectare))
deleteVariable(weegbreefonteinkruid_voortplanting_cluster)

write('ANB2019_Weegbreefonteinkruid2018.asc', IsNonZero(weegbreefonteinkruid_leefgebied))
deleteVariable(weegbreefonteinkruid_leefgebied)
