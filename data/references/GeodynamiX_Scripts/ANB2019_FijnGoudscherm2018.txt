// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(fijngoudscherm_voortplanting_bwk, bwk(1330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(fijngoudscherm_voortplanting_cluster, fuzzyClusterIdU(fijngoudscherm_voortplanting_bwk, 20, meter))

deleteVariable(fijngoudscherm_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(fijngoudscherm_leefgebied, cFilterGeArea(fijngoudscherm_voortplanting_cluster, 0.16, hectare))
deleteVariable(fijngoudscherm_voortplanting_cluster)

write('ANB2019_FijnGoudscherm2018.asc', IsNonZero(fijngoudscherm_leefgebied))
deleteVariable(fijngoudscherm_leefgebied)
