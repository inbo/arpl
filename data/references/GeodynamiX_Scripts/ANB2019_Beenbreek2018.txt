// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(beenbreek_voortplanting_bwk, bwk(4010%, 7110%, 7140%, ce%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(beenbreek_voortplanting_cluster, fuzzyClusterIdU(beenbreek_voortplanting_bwk, 20, meter))

deleteVariable(beenbreek_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(beenbreek_leefgebied, cFilterGeArea(beenbreek_voortplanting_cluster, 0.16, hectare))
deleteVariable(beenbreek_voortplanting_cluster)

write('ANB2019_Beenbreek2018.asc', IsNonZero(beenbreek_leefgebied))
deleteVariable(beenbreek_leefgebied)
