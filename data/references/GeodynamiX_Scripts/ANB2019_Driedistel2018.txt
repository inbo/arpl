// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(driedistel_voortplanting_bwk, bwk(2120%, 2130%, 2170%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(driedistel_voortplanting_cluster, fuzzyClusterIdU(driedistel_voortplanting_bwk, 20, meter))

deleteVariable(driedistel_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(driedistel_leefgebied, cFilterGeArea(driedistel_voortplanting_cluster, 0.16, hectare))
deleteVariable(driedistel_voortplanting_cluster)

write('ANB2019_Driedistel2018.asc', IsNonZero(driedistel_leefgebied))
deleteVariable(driedistel_leefgebied)
