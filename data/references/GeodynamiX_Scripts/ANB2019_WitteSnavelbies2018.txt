// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(wittesnavelbies_voortplanting_bwk, bwk(3160%, 4010%, 7140%, 7150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(wittesnavelbies_voortplanting_cluster, fuzzyClusterIdU(wittesnavelbies_voortplanting_bwk, 20, meter))

deleteVariable(wittesnavelbies_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(wittesnavelbies_leefgebied, cFilterGeArea(wittesnavelbies_voortplanting_cluster, 0.16, hectare))
deleteVariable(wittesnavelbies_voortplanting_cluster)

write('ANB2019_WitteSnavelbies2018.asc', IsNonZero(wittesnavelbies_leefgebied))
deleteVariable(wittesnavelbies_leefgebied)
