// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(alpenrus_voortplanting_bwk, bwk(7230%, mk, mk+, sm+))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(alpenrus_voortplanting_cluster, fuzzyClusterIdU(alpenrus_voortplanting_bwk, 20, meter))

deleteVariable(alpenrus_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(alpenrus_leefgebied, cFilterGeArea(alpenrus_voortplanting_cluster, 0.16, hectare))
deleteVariable(alpenrus_voortplanting_cluster)

write('ANB2019_Alpenrus2018.asc', IsNonZero(alpenrus_leefgebied))
deleteVariable(alpenrus_leefgebied)
