// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(rondezonnedauw_voortplanting_bwk, bwk(3160%, 4010%, 6230%, 7110%, 7120%, 7140%, 7150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(rondezonnedauw_voortplanting_cluster, fuzzyClusterIdU(rondezonnedauw_voortplanting_bwk, 20, meter))

deleteVariable(rondezonnedauw_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(rondezonnedauw_leefgebied, cFilterGeArea(rondezonnedauw_voortplanting_cluster, 0.16, hectare))
deleteVariable(rondezonnedauw_voortplanting_cluster)

write('ANB2019_RondeZonnedauw2018.asc', IsNonZero(rondezonnedauw_leefgebied))
deleteVariable(rondezonnedauw_leefgebied)
