// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(wijdbloeienderus_voortplanting_bwk, bwk(3130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(wijdbloeienderus_voortplanting_cluster, fuzzyClusterIdU(wijdbloeienderus_voortplanting_bwk, 20, meter))

deleteVariable(wijdbloeienderus_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(wijdbloeienderus_leefgebied, cFilterGeArea(wijdbloeienderus_voortplanting_cluster, 0.16, hectare))
deleteVariable(wijdbloeienderus_voortplanting_cluster)

write('ANB2019_WijdbloeiendeRus2018.asc', IsNonZero(wijdbloeienderus_leefgebied))
deleteVariable(wijdbloeienderus_leefgebied)
