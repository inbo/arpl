// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(liggendevleugeltjesbloem_voortplanting_bwk, bwk(6230%, rbbsg%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(liggendevleugeltjesbloem_voortplanting_cluster, fuzzyClusterIdU(liggendevleugeltjesbloem_voortplanting_bwk, 20, meter))

deleteVariable(liggendevleugeltjesbloem_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(liggendevleugeltjesbloem_leefgebied, cFilterGeArea(liggendevleugeltjesbloem_voortplanting_cluster, 0.16, hectare))
deleteVariable(liggendevleugeltjesbloem_voortplanting_cluster)

write('ANB2019_LiggendeVleugeltjesbloem2018.asc', IsNonZero(liggendevleugeltjesbloem_leefgebied))
deleteVariable(liggendevleugeltjesbloem_leefgebied)
