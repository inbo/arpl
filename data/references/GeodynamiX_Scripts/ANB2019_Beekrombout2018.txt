// clustering: indien geschikte cel binnen 10 meter, dan hoort die cel bij het waterbiotoop

// omgevend landbiotoop_bos
assign(beekrombout_landbiotoop_bos_bwk, bwk(ru%, 9160%, 91E0%, 91F0%, sf%, rbbsf%, sg%, rbbsg%, sk%, 6210%, sm%, 7140%, 7230%, 91E0%, so%, sp%, rbbsp%, sz%, v%, q%, 2180%, 9110%, 9120%, 9130%, 9150%, 9160%, 9190%, f%, l%, p%, rbbppm%, n, n+, n-, kp%))

deleteVariable(beekrombout_landbiotoop_bos_bwk)

// landbiotoop_bos moet minimaal 5 ha groot zijn
assign(beekrombout_landbiotoop_bos, cFilterGeArea(beekrombout_landbiotoop_bos_bwk_cluster, 5, hectare))

deleteVariable(beekrombout_landbiotoop_rest_bwk)

// landbiotoop_rest moet minimaal 50 ha groot zijn
assign(beekrombout_landbiotoop_rest, cFilterGeArea(beekrombout_landbiotoop_rest_bwk_cluster, 50, hectare))

// landbiotoop_bos moet deel uitmaken van landbiotoop_rest
assign(beekrombout_landbiotoop1, and(beekrombout_landbiotoop_rest, cellDistanceLe(beekrombout_landbiotoop_bos, 10, meter)))

assign(beekrombout_landbiotoop, and(beekrombout_landbiotoop1, cellDistanceLe(beekrombout_waterbiotoop1, 500, meter)))
