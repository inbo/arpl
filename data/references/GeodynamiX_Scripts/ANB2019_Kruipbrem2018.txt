// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kruipbrem_voortplanting_bwk, bwk(2310%, 4030%, 5130%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kruipbrem_voortplanting_cluster, fuzzyClusterIdU(kruipbrem_voortplanting_bwk, 20, meter))

deleteVariable(kruipbrem_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(kruipbrem_leefgebied, cFilterGeArea(kruipbrem_voortplanting_cluster, 0.16, hectare))
deleteVariable(kruipbrem_voortplanting_cluster)

write('ANB2019_Kruipbrem2018.asc', IsNonZero(kruipbrem_leefgebied))
deleteVariable(kruipbrem_leefgebied)
