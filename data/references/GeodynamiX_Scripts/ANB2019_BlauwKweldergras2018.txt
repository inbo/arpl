// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(blauwkweldergras_voortplanting_bwk, bwk(1330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(blauwkweldergras_voortplanting_cluster, fuzzyClusterIdU(blauwkweldergras_voortplanting_bwk, 20, meter))

deleteVariable(blauwkweldergras_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(blauwkweldergras_leefgebied, cFilterGeArea(blauwkweldergras_voortplanting_cluster, 0.16, hectare))
deleteVariable(blauwkweldergras_voortplanting_cluster)

write('ANB2019_BlauwKweldergras2018.asc', IsNonZero(blauwkweldergras_leefgebied))
deleteVariable(blauwkweldergras_leefgebied)
