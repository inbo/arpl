// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(dwergzegge_voortplanting_bwk, bwk(2190%, 7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(dwergzegge_voortplanting_cluster, fuzzyClusterIdU(dwergzegge_voortplanting_bwk, 20, meter))

deleteVariable(dwergzegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(dwergzegge_leefgebied, cFilterGeArea(dwergzegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(dwergzegge_voortplanting_cluster)

write('ANB2019_Dwergzegge2018.asc', IsNonZero(dwergzegge_leefgebied))
deleteVariable(dwergzegge_leefgebied)
