// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(harlekijn_voortplanting_bwk, bwk(2190%, 6230%, 6410%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(harlekijn_voortplanting_cluster, fuzzyClusterIdU(harlekijn_voortplanting_bwk, 20, meter))

deleteVariable(harlekijn_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(harlekijn_leefgebied, cFilterGeArea(harlekijn_voortplanting_cluster, 0.16, hectare))
deleteVariable(harlekijn_voortplanting_cluster)

write('ANB2019_Harlekijn2018.asc', IsNonZero(harlekijn_leefgebied))
deleteVariable(harlekijn_leefgebied)
