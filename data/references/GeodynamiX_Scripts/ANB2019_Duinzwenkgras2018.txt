// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(duinzwenkgras_voortplanting_bwk, bwk(2120%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(duinzwenkgras_voortplanting_cluster, fuzzyClusterIdU(duinzwenkgras_voortplanting_bwk, 20, meter))

deleteVariable(duinzwenkgras_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(duinzwenkgras_leefgebied, cFilterGeArea(duinzwenkgras_voortplanting_cluster, 0.16, hectare))
deleteVariable(duinzwenkgras_voortplanting_cluster)

write('ANB2019_Duinzwenkgras2018.asc', IsNonZero(duinzwenkgras_leefgebied))
deleteVariable(duinzwenkgras_leefgebied)
