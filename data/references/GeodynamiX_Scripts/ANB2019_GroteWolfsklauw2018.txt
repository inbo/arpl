// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(grotewolfsklauw_voortplanting_bwk, bwk(4030%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(grotewolfsklauw_voortplanting_cluster, fuzzyClusterIdU(grotewolfsklauw_voortplanting_bwk, 20, meter))

deleteVariable(grotewolfsklauw_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(grotewolfsklauw_leefgebied, cFilterGeArea(grotewolfsklauw_voortplanting_cluster, 0.16, hectare))
deleteVariable(grotewolfsklauw_voortplanting_cluster)

write('ANB2019_GroteWolfsklauw2018.asc', IsNonZero(grotewolfsklauw_leefgebied))
deleteVariable(grotewolfsklauw_leefgebied)
