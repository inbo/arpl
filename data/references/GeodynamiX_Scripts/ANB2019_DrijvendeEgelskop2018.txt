// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(drijvendeegelskop_voortplanting_bwk, bwk(3110%, 3160%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(drijvendeegelskop_voortplanting_cluster, fuzzyClusterIdU(drijvendeegelskop_voortplanting_bwk, 20, meter))

deleteVariable(drijvendeegelskop_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(drijvendeegelskop_leefgebied, cFilterGeArea(drijvendeegelskop_voortplanting_cluster, 0.16, hectare))
deleteVariable(drijvendeegelskop_voortplanting_cluster)

write('ANB2019_DrijvendeEgelskop2018.asc', IsNonZero(drijvendeegelskop_leefgebied))
deleteVariable(drijvendeegelskop_leefgebied)
