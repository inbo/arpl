assign(europesebever_houtig2, and(europesebever_houtig1, cellDistanceLe(europesebever_leefgebied1, 50, meter)))

deleteVariable(europesebever_houtig1)

assign(europesebever_leefgebied3, or(europesebever_leefgebied2, europesebever_houtig2))
