// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(rondezegge_voortplanting_bwk, bwk(7140%, 7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(rondezegge_voortplanting_cluster, fuzzyClusterIdU(rondezegge_voortplanting_bwk, 20, meter))

deleteVariable(rondezegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(rondezegge_leefgebied, cFilterGeArea(rondezegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(rondezegge_voortplanting_cluster)

write('ANB2019_RondeZegge2018.asc', IsNonZero(rondezegge_leefgebied))
deleteVariable(rondezegge_leefgebied)
