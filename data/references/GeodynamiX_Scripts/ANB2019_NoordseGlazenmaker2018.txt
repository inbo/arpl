// waterbiotoop (min 1 m˛): veenslenken en voedselarme wateren: Aom, Ao
// Niet: Ae, Aer, Kn, Ap, ad

// Direkte omgeving (10ha): bos (>50%) (so, sf, vm, vn, vo, vt, qb, qs, naaldhout alle types), natte heide /veen bij voorkeur met boomopslag (ce, cm, t), laagveen (ms, mk, mm), gagelstruweel (sm)
// Niet: droge heide, heischraal grasland (cg, cp, cd, cv, dm, ha), ruderale ruigte (ku, hr)

// Ruime omgeving (patch 100 ha)
// Hoofdzakelijk structuurrijk bos (alle struik en bos bwk eenheden) >70%

// waterbiotoop

assign(noordseglazenmaker_waterbiotoop_bwk1, bwk(ao, ao+, aoo, aoo+, 7110%, 3160%))
assign(noordseglazenmaker_waterbiotoop_notbwk, bwk(ae%, aer%, ap%, ad%, kn%))
assign(noordseglazenmaker_waterbiotoop_bwk, and(noordseglazenmaker_waterbiotoop_bwk1, not(noordseglazenmaker_waterbiotoop_notbwk)))

deleteVariable(noordseglazenmaker_waterbiotoop_bwk1)
deleteVariable(noordseglazenmaker_waterbiotoop_notbwk)

// clustering: indien geschikte cel binnen 50 meter, dan hoort die cel bij het waterbiotoop
assign(noordseglazenmaker_waterbiotoop_bwk_cluster, fuzzyClusterIdU(noordseglazenmaker_waterbiotoop_bwk, 50, meter))

// waterbiotoop moet minimaal 10 m˛ groot zijn
assign(noordseglazenmaker_waterbiotoop1, cFilterGeArea(noordseglazenmaker_waterbiotoop_bwk_cluster, 0.04, hectare))
//write('noordseglazenmaker_waterbiotoop1.asc', IsNonZero(noordseglazenmaker_waterbiotoop1))

deleteVariable(noordseglazenmaker_waterbiotoop_bwk)
deleteVariable(noordseglazenmaker_waterbiotoop_bwk_cluster)

// omgevend landbiotoop nabij

assign(noordseglazenmaker_landbiotoop_bwk1, bwk(so%, sf%, vm%, vn%, vo%, vt%, qb%, qs%, pa%, pi%, pm%, pp%, ceb%, cmb%, ct%, t%, ms%, mk%, mm%, sm%, 4010%))
assign(noordseglazenmaker_landbiotoop_notbwk, bwk(cg%, cp%, cd%, cv%, dm%, ha%, ku%, hr%))
assign(noordseglazenmaker_landbiotoop_bwk, and(noordseglazenmaker_landbiotoop_bwk1, not(noordseglazenmaker_landbiotoop_notbwk)))

// clustering: indien geschikte cel binnen 50 meter, dan hoort die cel bij het landbiotoop

assign(noordseglazenmaker_landbiotoop_bwk_cluster, fuzzyClusterIdU(noordseglazenmaker_landbiotoop_bwk, 50, meter))

deleteVariable(noordseglazenmaker_landbiotoop_bwk)

// landbiotoop moet minimaal 10 ha groot zijn

assign(noordseglazenmaker_landbiotoop1, cFilterGeArea(noordseglazenmaker_landbiotoop_bwk_cluster, 10, hectare))
//write('noordseglazenmaker_landbiotoop1.asc', IsNonZero(noordseglazenmaker_landbiotoop1))

deleteVariable(noordseglazenmaker_landbiotoop_bwk_cluster)

assign(noordseglazenmaker_landbiotoop_bos, bwk(so%, sf%, vm%, vn%, vo%, vt%, qb%, qs%, pa%, pi%, pm%, pp%))

// clustering: indien geschikte cel binnen 50 meter, dan hoort die cel bij het landbiotoop

assign(noordseglazenmaker_landbiotoop_bos_cluster, fuzzyClusterIdU(noordseglazenmaker_landbiotoop_bos, 50, meter))

deleteVariable(noordseglazenmaker_landbiotoop_bos)

// landbiotoop moet minimaal 5 ha groot zijn

assign(noordseglazenmaker_landbiotoop_bos1, cFilterGeArea(noordseglazenmaker_landbiotoop_bos_cluster, 5, hectare))
//write('noordseglazenmaker_landbiotoop_bos1.asc', IsNonZero(noordseglazenmaker_landbiotoop_bos1))

deleteVariable(noordseglazenmaker_landbiotoop_bos_cluster)
deleteVariable(noordseglazenmaker_landbiotoop_bwk1)
deleteVariable(noordseglazenmaker_landbiotoop_notbwk)

// landbiotoop_bos moet deel uitmaken van landbiotoop_rest
assign(noordseglazenmaker_landbiotoop_nabij, and(noordseglazenmaker_landbiotoop1, cellDistanceLe(noordseglazenmaker_landbiotoop_bos1, 10, meter)))
//write('noordseglazenmaker_landbiotoop_nabij.asc', IsNonZero(noordseglazenmaker_landbiotoop_nabij))

// omgevend landbiotoop ruimere omgeving

assign(noordseglazenmaker_landbiotoop_bwk_ruim, bwk(qb%, qs%, vn%, vm%, vo%, vt%, so%, ppm%, pm%, n, n-, n+, sf%, sm%, ceb%, mr%, mc%, mk%, mm%, ms%, hf%, cg%, ce%, cm%))

// clustering: indien geschikte cel binnen 50 meter, dan hoort die cel bij het landbiotoop

assign(noordseglazenmaker_landbiotoop_bwk_ruim_cluster, fuzzyClusterIdU(noordseglazenmaker_landbiotoop_bwk_ruim, 50, meter))

deleteVariable(noordseglazenmaker_landbiotoop_bwk_ruim)

// landbiotoop moet minimaal 35 ha groot zijn

assign(noordseglazenmaker_landbiotoop_ruim, cFilterGeArea(noordseglazenmaker_landbiotoop_bwk_ruim_cluster, 35, hectare))
//write('noordseglazenmaker_landbiotoop_ruim.asc', IsNonZero(noordseglazenmaker_landbiotoop_ruim))

deleteVariable(noordseglazenmaker_landbiotoop_bwk_ruim_cluster)

// landbiotoop2 moet in de buurt van waterbiotoop liggen

assign(noordseglazenmaker_waterbiotoop2, and(noordseglazenmaker_waterbiotoop1, cellDistanceLe(noordseglazenmaker_landbiotoop_nabij, 100, meter)))
//write('noordseglazenmaker_waterbiotoop2.asc', IsNonZero(noordseglazenmaker_waterbiotoop2))

assign(noordseglazenmaker_landbiotoop2, and(noordseglazenmaker_landbiotoop_nabij, cellDistanceLe(noordseglazenmaker_waterbiotoop1, 100, meter)))
//write('noordseglazenmaker_landbiotoop2.asc', IsNonZero(noordseglazenmaker_landbiotoop2))

deleteVariable(noordseglazenmaker_landbiotoop1)
deleteVariable(noordseglazenmaker_landbiotoop2)
deleteVariable(noordseglazenmaker_landbiotoop_nabij)
deleteVariable(noordseglazenmaker_waterbiotoop1)

assign(noordseglazenmaker_waterbiotoop, and(noordseglazenmaker_waterbiotoop2, cellDistanceLe(noordseglazenmaker_landbiotoop_ruim, 500, meter)))
//write('noordseglazenmaker_waterbiotoop.asc', IsNonZero(noordseglazenmaker_waterbiotoop))

assign(noordseglazenmaker_landbiotoop, and(noordseglazenmaker_landbiotoop_ruim, cellDistanceLe(noordseglazenmaker_waterbiotoop2, 100, meter)))
//write('noordseglazenmaker_landbiotoop.asc', IsNonZero(noordseglazenmaker_landbiotoop))

assign(noordseglazenmaker_leefgebied, or(noordseglazenmaker_waterbiotoop, noordseglazenmaker_landbiotoop))
write('ANB2019_NoordseGlazenmaker2018.asc', IsNonZero(noordseglazenmaker_leefgebied))

deleteVariable(noordseglazenmaker_landbiotoop)
deleteVariable(noordseglazenmaker_waterbiotoop)
deleteVariable(noordseglazenmaker_landbiotoop_ruim)
deleteVariable(noordseglazenmaker_waterbiotoop2)
