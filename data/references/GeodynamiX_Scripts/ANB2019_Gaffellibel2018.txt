// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(gaffellibel_voortplanting_bwk, bwk(3260%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(gaffellibel_voortplanting_cluster, fuzzyClusterIdU(gaffellibel_voortplanting_bwk, 20, meter))

deleteVariable(gaffellibel_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.04ha groot zijn

assign(gaffellibel_leefgebied, cFilterGeArea(gaffellibel_voortplanting_cluster, 0.04, hectare))
deleteVariable(gaffellibel_voortplanting_cluster)

write('ANB2019_Gaffellibel2018.asc', IsNonZero(gaffellibel_leefgebied))
deleteVariable(gaffellibel_leefgebied)
