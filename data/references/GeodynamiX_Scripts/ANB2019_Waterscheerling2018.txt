// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(waterscheerling_voortplanting_bwk, bwk(7140%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(waterscheerling_voortplanting_cluster, fuzzyClusterIdU(waterscheerling_voortplanting_bwk, 20, meter))

deleteVariable(waterscheerling_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(waterscheerling_leefgebied, cFilterGeArea(waterscheerling_voortplanting_cluster, 0.16, hectare))
deleteVariable(waterscheerling_voortplanting_cluster)

write('ANB2019_Waterscheerling2018.asc', IsNonZero(waterscheerling_leefgebied))
deleteVariable(waterscheerling_leefgebied)
