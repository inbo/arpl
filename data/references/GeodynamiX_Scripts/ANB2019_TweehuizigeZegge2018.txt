// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(tweehuizigezegge_voortplanting_bwk, bwk(7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(tweehuizigezegge_voortplanting_cluster, fuzzyClusterIdU(tweehuizigezegge_voortplanting_bwk, 20, meter))

deleteVariable(tweehuizigezegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(tweehuizigezegge_leefgebied, cFilterGeArea(tweehuizigezegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(tweehuizigezegge_voortplanting_cluster)

write('ANB2019_TweehuizigeZegge2018.asc', IsNonZero(tweehuizigezegge_leefgebied))
deleteVariable(tweehuizigezegge_leefgebied)
