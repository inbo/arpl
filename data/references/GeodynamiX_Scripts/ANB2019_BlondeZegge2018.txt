// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(blondezegge_voortplanting_bwk, bwk(6410%, 7230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(blondezegge_voortplanting_cluster, fuzzyClusterIdU(blondezegge_voortplanting_bwk, 20, meter))

deleteVariable(blondezegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(blondezegge_leefgebied, cFilterGeArea(blondezegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(blondezegge_voortplanting_cluster)

write('ANB2019_BlondeZegge2018.asc', IsNonZero(blondezegge_leefgebied))
deleteVariable(blondezegge_leefgebied)
