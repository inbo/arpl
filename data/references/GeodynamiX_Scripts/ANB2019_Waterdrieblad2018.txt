// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(waterdrieblad_voortplanting_bwk, bwk(3160%, 7140%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(waterdrieblad_voortplanting_cluster, fuzzyClusterIdU(waterdrieblad_voortplanting_bwk, 20, meter))

deleteVariable(waterdrieblad_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(waterdrieblad_leefgebied, cFilterGeArea(waterdrieblad_voortplanting_cluster, 0.16, hectare))
deleteVariable(waterdrieblad_voortplanting_cluster)

write('ANB2019_Waterdrieblad2018.asc', IsNonZero(waterdrieblad_leefgebied))
deleteVariable(waterdrieblad_leefgebied)
