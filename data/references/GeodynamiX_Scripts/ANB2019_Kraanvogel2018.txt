// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kraanvogel_voortplanting_bwk, bwk(4010%, 7%, md%, mm%, mr%, ms%, mz%, c%, so%, sm%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kraanvogel_voortplanting_cluster, fuzzyClusterIdU(kraanvogel_voortplanting_bwk, 50, meter))

deleteVariable(kraanvogel_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 500ha groot zijn

assign(kraanvogel_leefgebied, cFilterGeArea(kraanvogel_voortplanting_cluster, 750, hectare))
deleteVariable(kraanvogel_voortplanting_cluster)

write('ANB2019_Kraanvogel2018.asc', IsNonZero(kraanvogel_leefgebied))
deleteVariable(kraanvogel_leefgebied)
