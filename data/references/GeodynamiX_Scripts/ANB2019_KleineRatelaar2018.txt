// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kleineratelaar_voortplanting_bwk, bwk(2190%, 6120%, 6230%, 6510%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kleineratelaar_voortplanting_cluster, fuzzyClusterIdU(kleineratelaar_voortplanting_bwk, 20, meter))

deleteVariable(kleineratelaar_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(kleineratelaar_leefgebied, cFilterGeArea(kleineratelaar_voortplanting_cluster, 0.16, hectare))
deleteVariable(kleineratelaar_voortplanting_cluster)

write('ANB2019_KleineRatelaar2018.asc', IsNonZero(kleineratelaar_leefgebied))
deleteVariable(kleineratelaar_leefgebied)
