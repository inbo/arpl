// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(kwelderzegge_voortplanting_bwk, bwk(1330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(kwelderzegge_voortplanting_cluster, fuzzyClusterIdU(kwelderzegge_voortplanting_bwk, 20, meter))

deleteVariable(kwelderzegge_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(kwelderzegge_leefgebied, cFilterGeArea(kwelderzegge_voortplanting_cluster, 0.16, hectare))
deleteVariable(kwelderzegge_voortplanting_cluster)

write('ANB2019_Kwelderzegge2018.asc', IsNonZero(kwelderzegge_leefgebied))
deleteVariable(kwelderzegge_leefgebied)
