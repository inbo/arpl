// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(rodedophei_voortplanting_bwk, bwk(4030%, 5130%, 6230%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(rodedophei_voortplanting_cluster, fuzzyClusterIdU(rodedophei_voortplanting_bwk, 20, meter))

deleteVariable(rodedophei_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(rodedophei_leefgebied, cFilterGeArea(rodedophei_voortplanting_cluster, 0.16, hectare))
deleteVariable(rodedophei_voortplanting_cluster)

write('ANB2019_RodeDophei2018.asc', IsNonZero(rodedophei_leefgebied))
deleteVariable(rodedophei_leefgebied)
