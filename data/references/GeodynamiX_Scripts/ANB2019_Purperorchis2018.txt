// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(purperorchis_voortplanting_bwk, bwk(6210%, 9150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(purperorchis_voortplanting_cluster, fuzzyClusterIdU(purperorchis_voortplanting_bwk, 20, meter))

deleteVariable(purperorchis_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(purperorchis_leefgebied, cFilterGeArea(purperorchis_voortplanting_cluster, 0.16, hectare))
deleteVariable(purperorchis_voortplanting_cluster)

write('ANB2019_Purperorchis2018.asc', IsNonZero(purperorchis_leefgebied))
deleteVariable(purperorchis_leefgebied)
