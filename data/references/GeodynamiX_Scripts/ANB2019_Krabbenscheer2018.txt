// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(krabbenscheer_voortplanting_bwk, bwk(3150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(krabbenscheer_voortplanting_cluster, fuzzyClusterIdU(krabbenscheer_voortplanting_bwk, 20, meter))

deleteVariable(krabbenscheer_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(krabbenscheer_leefgebied, cFilterGeArea(krabbenscheer_voortplanting_cluster, 0.16, hectare))
deleteVariable(krabbenscheer_voortplanting_cluster)

write('ANB2019_Krabbenscheer2018.asc', IsNonZero(krabbenscheer_leefgebied))
deleteVariable(krabbenscheer_leefgebied)
