// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(tauvlinder_voortplanting_bwk, bwk(9110%, 9120%, 9130%, 9150%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(tauvlinder_voortplanting_cluster, fuzzyClusterIdU(tauvlinder_voortplanting_bwk, 20, meter))

deleteVariable(tauvlinder_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 10ha groot zijn

assign(tauvlinder_leefgebied, cFilterGeArea(tauvlinder_voortplanting_cluster, 2, hectare))
deleteVariable(tauvlinder_voortplanting_cluster)

// baken ecoregio van ecodistrictgebieden af
assign(tauvlinder_ecodistrict, OrEq('ecoregio.asc', 27, 28))
assign(tauvlinder_leefgebied_ecodistrict, and(tauvlinder_leefgebied, tauvlinder_ecodistrict))

deleteVariable(tauvlinder_leefgebied)
deleteVariable(tauvlinder_ecodistrict)

write('ANB2019_tauvlinder2018.asc', IsNonZero(tauvlinder_leefgebied_ecodistrict))

deleteVariable(tauvlinder_leefgebied_ecodistrict)
