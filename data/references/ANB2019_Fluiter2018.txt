// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(fluiter_voortplanting_bwk, bwk(9110%, 9120%, 9130%, 9160%, 9190%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(fluiter_voortplanting_cluster, fuzzyClusterIdU(fluiter_voortplanting_bwk, 20, meter))

deleteVariable(fluiter_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 5ha groot zijn

assign(fluiter_leefgebied, cFilterGeArea(fluiter_voortplanting_cluster, 5, hectare))
deleteVariable(fluiter_voortplanting_cluster)

write('ANB2019_Fluiter2018.asc', IsNonZero(fluiter_leefgebied))
deleteVariable(fluiter_leefgebied)
