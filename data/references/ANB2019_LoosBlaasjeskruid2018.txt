// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(loosblaasjeskruid_voortplanting_bwk, bwk(3150%, 3160%, 7140%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(loosblaasjeskruid_voortplanting_cluster, fuzzyClusterIdU(loosblaasjeskruid_voortplanting_bwk, 20, meter))

deleteVariable(loosblaasjeskruid_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(loosblaasjeskruid_leefgebied, cFilterGeArea(loosblaasjeskruid_voortplanting_cluster, 0.16, hectare))
deleteVariable(loosblaasjeskruid_voortplanting_cluster)

write('ANB2019_LoosBlaasjeskruid2018.asc', IsNonZero(loosblaasjeskruid_leefgebied))
deleteVariable(loosblaasjeskruid_leefgebied)
