// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(gewonezoutmelde_voortplanting_bwk, bwk(1130%, 1310%, 1330%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(gewonezoutmelde_voortplanting_cluster, fuzzyClusterIdU(gewonezoutmelde_voortplanting_bwk, 20, meter))

deleteVariable(gewonezoutmelde_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(gewonezoutmelde_leefgebied, cFilterGeArea(gewonezoutmelde_voortplanting_cluster, 0.16, hectare))
deleteVariable(gewonezoutmelde_voortplanting_cluster)

write('ANB2019_GewoneZoutmelde2018.asc', IsNonZero(gewonezoutmelde_leefgebied))
deleteVariable(gewonezoutmelde_leefgebied)
