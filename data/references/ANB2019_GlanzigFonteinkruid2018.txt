// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(glanzigfonteinkruid_voortplanting_bwk, bwk(3150%, 3260%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(glanzigfonteinkruid_voortplanting_cluster, fuzzyClusterIdU(glanzigfonteinkruid_voortplanting_bwk, 20, meter))

deleteVariable(glanzigfonteinkruid_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(glanzigfonteinkruid_leefgebied, cFilterGeArea(glanzigfonteinkruid_voortplanting_cluster, 0.16, hectare))
deleteVariable(glanzigfonteinkruid_voortplanting_cluster)

write('ANB2019_GlanzigFonteinkruid2018.asc', IsNonZero(glanzigfonteinkruid_leefgebied))
deleteVariable(glanzigfonteinkruid_leefgebied)
