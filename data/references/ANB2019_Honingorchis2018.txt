// baken de voortplantingsgebieden af (zonder opgespoten terreinen)

assign(honingorchis_voortplanting_bwk, bwk(2190%))

// clustering: indien geschikte cel binnen 20 meter, dan hoort die cel bij het voortplantingscluster

assign(honingorchis_voortplanting_cluster, fuzzyClusterIdU(honingorchis_voortplanting_bwk, 20, meter))

deleteVariable(honingorchis_voortplanting_bwk)

// optimale voortplantingsclusters moeten minimaal 0.16ha groot zijn

assign(honingorchis_leefgebied, cFilterGeArea(honingorchis_voortplanting_cluster, 0.16, hectare))
deleteVariable(honingorchis_voortplanting_cluster)

write('ANB2019_Honingorchis2018.asc', IsNonZero(honingorchis_leefgebied))
deleteVariable(honingorchis_leefgebied)
